거듭제곱하는 방법
--------------------------
암기해야할 예제
--------------------------
(1)반복문을 이용하기
def power_of_2_loop( k ): # Output: 2k
	i = 0
	power = 1
	while i < k:
		power = power * 2
		i+=1
	return power
print( power_of_2_loop(3) )

(2)재귀호출 이용하기
def power_of_2_recur(k): # Output: 2k
    if k == 0:
        return 1
    else:
        return 2 * power_of_2_recur(k - 1)
print( power_of_2_recur(3) )
